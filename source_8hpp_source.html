25 /*
26  * Base class for sources
27  *
28  * This class is the base class for all sources. It defines the interface for
29  * providing output of data internally acquired.
30  * Child classes must implement the kind() and get_output() methods.
31  * Optionally, they can implement the set_params() method to receive parameters
32  * as a void pointer.
33  *
34  * @tparam Tout Output data type
35  */
36 template <typename Tout = std::vector<double>>
37 class Source {
38 public:
39  Source() : _error("No error") , _blob_format("none") {}
40  virtual ~Source() {}
41 
42  /*
43  * Returns the kind of source
44  *
45  * This method returns the kind of source. It is used to identify the source
46  * when loading it from a plugin.
47  *
48  * @return The kind of source
49  */
50  virtual std::string kind() = 0;
51 
52  /*
53  * Get the output data
54  *
55  * This method provides the output data. It returns
56  * true if the data was fetched successfully, and false otherwise.
57  *
58  * @param out The output data
59  * @return True if the data was processed successfully, and false otherwise
60  */
61  virtual return_type get_output(Tout *out, std::vector<unsigned char> *blob = nullptr) = 0;

63 
64  /*
65  * Sets the parameters

67  * This method sets the parameters for the source. It receives a void pointer

72  */
